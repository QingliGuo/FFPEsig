df_reconstructed = pd.DataFrame()
df_reconstructed['53C_Concordant'] = FFPE_profiles['53C_Concordant'][reverse_T2C_ind].copy()

for i, sample in enumerate(original_profiles.columns[1:5]):
    v = original_profiles[sample].to_numpy()
    npr.seed(123)

    if "unrepaired" in sample:
        w, h, loss = sig_extraction(V = v.reshape(len(v),1), W1 = ffpe_sig_unrepaired, precision = 0.98)
    else:
        w, h, loss = sig_extraction(V = v.reshape(len(v),1), W1 = ffpe_sig_repaired, precision = 0.98)
        
    reconstructed_tmp = (w[:,1]*h[1]).astype("int")    
    df_reconstructed[sample + '_Corrected'] = reconstructed_tmp[reverse_T2C_ind].copy()
    
    if i == 1:
        df_reconstructed[''] = [0] * sum(reverse_T2C_ind)
        df_reconstructed['C277_Concordant'] = FFPE_profiles['C277_Concordant'][reverse_T2C_ind].copy()
